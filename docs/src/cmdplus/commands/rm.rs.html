5/// Remove a file or directory.
6///
7/// ## English
8/// This function removes a file or directory. If the `recursive` flag is set, it will remove a directory and all of its contents.
9/// If the `dir` flag is set, it will only remove empty directories.
10///
11/// ## 日本語
12/// この関数はファイルまたはディレクトリを削除します。  
13/// `recursive` フラグが設定されている場合、ディレクトリとその内容を再帰的に削除します。  
14/// `dir` フラグが設定されている場合、空のディレクトリのみを削除します。
15pub fn rm_command(path: &str, recursive: bool, dir: bool) -> io::Result<()> {
16    let path = Path::new(path);
17
18    // Check if it's a directory and handle the flags accordingly
19    if path.is_dir() {
20        if recursive {
21            // If recursive, remove directory and all its contents
22            remove_dir_all(path)
23        } else if dir {
24            // If dir flag, ensure the directory is empty before removal
25            if fs::read_dir(path)?.count() == 0 {
26                remove_dir(path)
27            } else {
28                Err(io::Error::new(ErrorKind::Other, "Directory is not empty"))
29            }
30        } else {
31            // If it's a directory but no flags are set, return an error
32            Err(io::Error::new(
33                ErrorKind::Other,
34                "Directory removal requires --recursive or --dir",
35            ))
36        }
37    } else if path.is_file() {
38        // If it's a file, simply remove it
39        remove_file(path)
40    } else {
41        // If path is neither a file nor directory, return an error
42        Err(io::Error::new(
43            ErrorKind::NotFound,
44            "File or directory not found",
45        ))
46    }
47}

49/// Recursively remove a directory and its contents.
50///
51/// ## English
52/// This function recursively removes a directory and all its contents.
53///
54/// ## 日本語
55/// この関数はディレクトリとその中身を再帰的に削除します。
56fn remove_dir_all(path: &Path) -> io::Result<()> {
57    // Iterate through all files and subdirectories in the directory
58    for entry in fs::read_dir(path)? {
59        let entry = entry?;
60        let entry_path = entry.path();
61        if entry_path.is_dir() {
62            // Recursively remove subdirectories
63            remove_dir_all(&entry_path)?;
64        } else {
65            // Remove files
66            remove_file(entry_path)?;
67        }
68    }
69
70    // Finally, remove the empty directory
71    remove_dir(path)
72}
