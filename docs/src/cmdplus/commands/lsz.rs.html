1//! # `lsz` command implementation
2//!
3//! ## English
4//! This module implements the `lsz` command.
5//! It displays the size of all files and directories within a given directory.
6//!
7//! ## 日本語
8//! このモジュールは `lsz` コマンドを実装しています。  
9//! 指定されたディレクトリ内のすべてのファイルとディレクトリのサイズを表示します。

11use std::fs;
12use std::path::Path;
13
14/// ## English
15/// Calculate and return the size of all files and subdirectories within a given directory.
16/// Returns a formatted string including each entry and the total size.
17///
18/// ## 日本語
19/// 指定されたディレクトリ内のファイルとサブディレクトリのサイズを合計し、  
20/// 一覧表示形式の文字列で返します。
21pub fn lsz_command(path: &str) -> Result<String, std::io::Error> {
22    let path = Path::new(path);
23    let mut output = String::new();
24    let mut total_size = 0u64;
25
26    if path.is_dir() {
27        for entry in fs::read_dir(path)? {
28            let entry = entry?;
29            let metadata = entry.metadata()?;
30            let file_size = if metadata.is_dir() {
31                calculate_dir_size(&entry.path())?
32            } else {
33                metadata.len()
34            };
35
36            total_size += file_size;
37
38            output.push_str(&format!(
39                "{}\t{} bytes\n",
40                entry.file_name().to_string_lossy(),
41                file_size
42            ));
43        }
44
45        output.push_str(&format!("Total size: {} bytes\n", total_size));
46        Ok(output)
47    } else {
48        Err(std::io::Error::new(
49            std::io::ErrorKind::InvalidInput,
50            "Not a directory",
51        ))
52    }
53}
54
55/// ## English
56/// Recursively calculates the total size of a directory and its contents.
57///
58/// ## 日本語
59/// ディレクトリ内のすべてのファイルおよびサブディレクトリのサイズを  
60/// 再帰的に合計して返します。
61fn calculate_dir_size(path: &Path) -> Result<u64, std::io::Error> {
62    let mut size = 0;
63
64    for entry in fs::read_dir(path)? {
65        let entry = entry?;
66        let metadata = entry.metadata()?;
67        if metadata.is_dir() {
68            size += calculate_dir_size(&entry.path())?;
69        } else {
70            size += metadata.len();
71        }
72    }
73
74    Ok(size)
75}
