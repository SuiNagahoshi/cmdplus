1//! # `lsz` command implementation
2//!
3//! ## English
4//! This module implements the `lsz` command.
5//! It lists the total size of each directory and file under the given path.
6//!
7//! ## 日本語
8//! このモジュールは `lsz` コマンドを実装しています。  
9//! 指定したパス以下のファイルおよびディレクトリの合計サイズを一覧表示します。
10
11use std::fs;
12use std::io;
13use std::path::{Path, PathBuf};
14
15/// List directory and file sizes recursively.
16///
17/// ## English
18/// Walks through the target directory and prints size of files and directories.
19///
20/// ## 日本語
21/// 指定ディレクトリ以下を再帰的に走査し、  
22/// 各ファイル・ディレクトリのサイズを表示します。
23pub fn lsz_command(dir: &str) -> io::Result<()> {
24    let path = Path::new(dir);
25    let mut entries = Vec::new();
26    collect_sizes(path, &mut entries)?;
27
28    entries.sort_by(|a, b| b.1.cmp(&a.1)); // Sort descending by size
29
30    for (path, size) in entries {
31        println!("{:>10}  {}", format_size(size), path.display());
32    }
33
34    Ok(())
35}
36
37/// Collect sizes of files and directories recursively.
38///
39/// ## English
40/// Populates a vector with the total size for each item.
41///
42/// ## 日本語
43/// ファイルとディレクトリの合計サイズを計算してベクタに格納します。
44fn collect_sizes(path: &Path, entries: &mut Vec<(PathBuf, u64)>) -> io::Result<u64> {
45    let metadata = fs::symlink_metadata(path)?;
46    if metadata.is_file() {
47        let size = metadata.len();
48        entries.push((path.to_path_buf(), size));
49        Ok(size)
50    } else if metadata.is_dir() {
51        let mut total = 0;
52        for entry in fs::read_dir(path)? {
53            let entry = entry?;
54            total += collect_sizes(&entry.path(), entries)?;
55        }
56        entries.push((path.to_path_buf(), total));
57        Ok(total)
58    } else {
59        Ok(0)
60    }
61}

63/// Format size in bytes to a readable string.
64///
65/// ## English
66/// Converts bytes to a human-readable format like KB, MB, etc.
67///
68/// ## 日本語
69/// バイト数を KB や MB などの読みやすい形式に変換します。
70fn format_size(size: u64) -> String {
71    const UNITS: [&str; 5] = ["B", "KB", "MB", "GB", "TB"];
72    let mut size = size as f64;
73    let mut unit = 0;
74
75    while size >= 1024.0 && unit < UNITS.len() - 1 {
76        size /= 1024.0;
77        unit += 1;
78    }
79
80    format!("{:.1}{}", size, UNITS[unit])
81}
